index.html

Including the relevant javascript files. For the sake of clarity, every logical component will be separated into a specific file

Example of index.html file

Here is a simple example of an index.html page containing the _UT library and the order in which the plugins should be included.

<html>
<head>

<!-- ... HEAD CONTENT -->

</head>
<body>

<!-- ... BODY CONTENT -->

  <!-- PAGE VARIABLES -->
  <script type="text/javascript">
    s_pagename = "home";
    s_display_language = "en";
    s_section = "other";
  </script>

  <!-- Including the Main _UT library -->
  <script type="text/javascript" src="js/_UT.js"></script>

  <!-- Including Adapters for Omniture and Google Analytics -->
  <script type="text/javascript" src="js/gaAdapter.js"></script>
  <script type="text/javascript" src="js/omniAdapter.js"></script>

  <!-- Including common settings and common Extensions to the UT object -->
  <script type="text/javascript" src="js/settings/_UT.extensions.js"></script>
  <script type="text/javascript" src="js/settings/common.js"></script>

  <!-- Including the Settings/Business Logic and Mapping for each Adapter Instance --->
  <script type="text/javascript" src="js/settings/omniSettings.js"></script>
  <script type="text/javascript" src="js/settings/gaSettings.js"></script>

  <!-- Creating a Container to manage the adapters and fire the beacons --->
  <script type="text/javascript" src="js/settings/container.js"></script>

</body>
</html>

UT.extensions.js

This file contains _UT Object extensions which will be available globally in the _UT namespace.

Example of UT.extensions.js file

In this example we are adding the getQueryParam and toLowerCase static functions to the _UT namespace.

//*************************************************
// *    Non-Core utilities - affects _UT Static Object
//****************************************************
_UT.addExtensions({
   
    getQueryParam: function (name) {
        var names_arr = name.split(",");
        if(names_arr.length > 1) {
            for(var i=0; i<names_arr.length;i++){
                if((r = _UT.getQueryParam(names_arr[i])) !== "") return r;
            }
            return "";
        }
        else {
            name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
            var regexS = "[\\?&]" + name + "=([^&#]*)";
            var regex = new RegExp(regexS);
            var results = regex.exec(window.location.href);
            if (results == null) return "";
            else return decodeURIComponent(results[1].replace(/\+/g, " "));
        }
    },
    
    /** To Lower Case
     *
     *  @param object    The object where each value will be lowerCased
     * 
     *  @return object
     */    
     lowerCase: function(obj) {
         _UT.each(obj, function(i,v) {
            if(typeof v == "string") obj[i] = v.toLowerCase();
        });
         return obj;
     }

});

common.js

This file contains variables and settings that are shared by all Adapters; the processor is one such Object that is of particular importance if all Adapters will use the same business logic.

Example of common.js file

			s_dev_urls = base_dev_urls + "," + settings.s_dev_urls;

		if(typeof settings.s_omni_prod_account != "undefined")

			s_ga_dev_account = settings.s_ga_dev_account;

		break;
	}
	
}

function s_domainMatches(list) {
	var cur_domain = window.location.hostname;
	var domains = list.split(",");
	for(var i=0;i<domains.length;i++) {
		if(cur_domain.indexOf(domains[i]) != -1) return true;
	}
	return false;
}

container.js

The container file puts all the previous settings together to fire all the adapters

Example of container.js file

The first part of the code serves to load the settings of each adapter instance into a container object and then call the fire function of each. An example is shown of how to use the container to fire for only a single instance using that adapter's id.

//###########################################################################
// UT CONTAINER
//###########################################################################

    // using container to create the objects
    _ut_container = new _UT.Container(
		{settings: OMNI_SETTINGS, map: OMNI_MAP, processor: OMNI_PROCESSOR}
		,{settings: GA_SETTINGS, map: GA_MAP, processor: GA_PROCESSOR}
	);
	
	// using container to fire
	_ut_container.call('fire',false,{
		__usePrevious: false,
		__useWindow: true
	});
	
//###########################################################################
// Additional Functions
//###########################################################################

	function s_trackPageview(params) {
		if(typeof params.__usePrevious == "undefined") params.__usePrevious = false;
		if(typeof params.__useWindow == "undefined") params.__useWindow = true;
		_ut_container.call('fire',
				 {__include: ['_ut_omni','_ut_ga']},
				 params
		);
	}
